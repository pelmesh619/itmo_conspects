        \Ex –ê—Ä–∏—Ñ–º–µ—Ç–∏—á–µ—Å–∫–∞—è –ø—Ä–æ–≥—Ä–µ—Å—Å–∏—è

        $a_n = \begin{cases}a_0 = const \quad n = 0 \\ a_{n - 1} + d, \quad n > 0\end{cases}$

        –†–µ—à–µ–Ω–∏–µ: $a_n = a_0 + nd$ -- –∞–Ω–∑–∞—Ü (Ansatz, –¥–æ–≥–∞–¥–∫–∞)

        –ü—Ä–æ–≤–µ—Ä–∫–∞: $a_n = a_0 + nd = a_{n - 1} + d = a_0 + (n - 1)d + d = a_0 + nd \quad$ - {\Largeüëçüëç}

        \item –ú–µ—Ç–æ–¥ —Ö–∞—Ä–∞–∫—Ç–µ—Ä–∏—Å—Ç–∏—á–µ—Å–∫–æ–≥–æ —É—Ä–∞–≤–Ω–µ–Ω–∏—è

        \substack{\text{–†–µ–∫—É—Ä—Ä–µ–Ω—Ç–Ω–æ–µ} \\ \text{—Å–æ–æ—Ç–Ω–æ—à–µ–Ω–∏–µ}} \ $\stackrel{a_n \to r^n}{\rightsquigarrow}$ \ \substack{\text{–•–∞—Ä–∞–∫—Ç–µ—Ä–∏—Å—Ç–∏—á–µ—Å–∫–æ–µ} \\ \text{—É—Ä–∞–≤–Ω–µ–Ω–∏–µ}} \ $\stackrel{\text{—Ä–µ—à–µ–Ω–∏–µ}}{\rightsquigarrow}$ –ö–æ—Ä–Ω–∏ $\stackrel{\text{–º–∞–≥–∏—è}}{\rightsquigarrow}$ –†–µ—à–µ–Ω–∏–µ $\rightsquigarrow$ –ü—Ä–æ–≤–µ—Ä–∫–∞

        \Ex $a_n = a_{n - 1} + 6a_{n - 2}$

        $r^n - r^{n - 1} - 6r^{n - 2} = 0$

        \item \textbf{–†–∞–∑–¥–µ–ª—è–π –∏ –≤–ª–∞—Å—Ç–≤—É–π} (Divide-and-Conquer)

        \hypertarget{mastertheorem}{}

        \hfill\href{https://ru.wikipedia.org/wiki/%D0%9E%D1%81%D0%BD%D0%BE%D0%B2%D0%BD%D0%B0%D1%8F_%D1%82%D0%B5%D0%BE%D1%80%D0%B5%D0%BC%D0%B0_%D0%BE_%D1%80%D0%B5%D0%BA%D1%83%D1%80%D1%80%D0%B5%D0%BD%D1%82%D0%BD%D1%8B%D1%85_%D1%81%D0%BE%D0%BE%D1%82%D0%BD%D0%BE%D1%88%D0%B5%D0%BD%D0%B8%D1%8F%D1%85}{*—Ç—ã–∫*}


        $T(n) = aT\left(\frac{n}{b}\right) + f(n)$

        –ò–∑ —ç—Ç–æ–≥–æ, $c_{crit} = \log_b a$

        \mediumvspace

        \underline{I —Å–ª—É—á–∞–π}: —Å–ª–∏—è–Ω–∏–µ $<$ —Ä–µ–∫—É—Ä—Å–∏—è

        $f(n) \in O(n^c)$, –≥–¥–µ $c < c_{crit} \Longrightarrow T(n) \in \Theta(n^{c_{crit}})$

        $f(n) \in O(n^c) \Longleftrightarrow f(n) \in o(n^{c_{crit}})$

        \mediumvspace

        \underline{II —Å–ª—É—á–∞–π}: —Å–ª–∏—è–Ω–∏–µ $\approx$ —Ä–µ–∫—É—Ä—Å–∏—è

        $f(n) \in \Omega(n^c)$, –≥–¥–µ $c > c_{crit} \Longrightarrow T(n) \in \Theta(f(n))$

        \hypertarget{akrabazzimethod}{}

        \item \textbf{–ú–µ—Ç–æ–¥ –ê–∫—Ä–∞-–ë–∞—Ü—Ü–∏} (Akra-Bazzi method)
        \hfill\href{https://en.wikipedia.org/wiki/Akra%E2%80%93Bazzi_method}{*—Ç—ã–∫*}


        $T(n) = f(n) + \sum_{i = 1}^k a_i T(b_i n + h_i(n)) \Longrightarrow T(n) \in \Theta\left(n^p \cdot \left(1 + \int_1^n \frac{f(x)}{x^{p + 1}} dx\right)\right)$, –≥–¥–µ $p$ -- —Ä–µ—à–µ–Ω–∏–µ –¥–ª—è $\sum_{i = 1}^k a_i b_i^p = 1$

        \begin{cases}
            k = const \\
            a_i > 0 \\
            0 < b_i < 1 \\
            h_1(n) \in O\left(\frac{n}{\log^2 n}\right) \text{ -- –º–∞–ª—ã–µ –≤–æ–∑–º—É—â–µ–Ω–∏—è}
        \end{cases}

        \Ex $T(n) = T\left(\left\lfloor\frac{n}{2}\right\rfloor\right) + T\left(\left\lceil\frac{n}{2}\right\rceil\right) + n$ -- –∞—Å–∏–º–ø—Ç–æ—Ç–∏–∫–∞ —Å–æ—Ä—Ç–∏—Ä–æ–≤–∫–∏ —Å–ª–∏—è–Ω–∏–µ–º

        –ó–¥–µ—Å—å $b_i = \frac{1}{2}, \quad h = \pm O(1) \in O\left(\frac{n}{\log^2 n}\right)$

        \Ex $T(n) = T\left(\frac{3n}{4}\right) + T\left(\frac{n}{4}\right) + n$

        $a_1 = 1, b_1 = \frac{3}{4}, a_2 = 1, b_2 = \frac{1}{4}, f(n) = n$

        $\left(\frac{3}{4}\right)^p + \left(\frac{1}{4}\right)^p = 1$

        $p = 1$

        $\int_1^n \frac{x}{x^{1 + 1}}dx = \int_1^n \frac{dx}{x} = \ln x \Big|_1^n = \ln n$

        $T(n) \in \Theta(n \cdot (1 + \ln n))$

        $T(n) \in \Theta(n \ln n)$
