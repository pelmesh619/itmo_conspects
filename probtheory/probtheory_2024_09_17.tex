        P(A_1) P(A_2 | A_1) P(A_3 | A_1 A_2) \dots P(A_n | A_1 A_2 \dots A_{n - 1})$

        $\Box$

        $P(A) = P(\Omega A) = P((H_1 + H_2 + H_3 + \dots) A) = P(H_1 A + H_2 A + H_3 A + \dots) = [H_i \cdot A \cdot H_j \cdot A = \emptyset \cdot A] = P(H_1 A) + P(H_2 A) + \dots =
        P(H_1) P(A | H_1) + P(H_2) P(A | H_2) + \dots$

        $\Box$
    \end{tcolorbox}

    \begin{MyTheorem}
        \ThNs{–§–æ—Ä–º—É–ª–∞ –ë–∞–π–µ—Å–∞} $\letsymbol H_1, H_2, \dots, H_n$ - –ø–æ–ª–Ω–∞—è –≥—Ä—É–ø–ø–∞ —Å–æ–±—ã—Ç–∏–π, –∏ –∏–∑–≤–µ—Å—Ç–Ω–æ, —á—Ç–æ —Å–æ–±—ã—Ç–∏–µ $A$ —É–∂–µ –ø—Ä–æ–∏–∑–æ—à–ª–æ

        –¢–æ–≥–¥–∞ $P(H_k | A) = \frac{P(H_k) P(A | H_k)}{\sum_{i = 1}^\infty P(H_i) P(A | H_i)}$
    \end{MyTheorem}

    \begin{tcolorbox}
        $\Box$

        $P(H_k | A) = \frac{P(H_k A)}{P(A)} = \frac{P(H_k) P(A|H_k)}{\sum_{i = 1}^\infty P(H_i) P(A|H_i)}$

        $\Box$
    \end{tcolorbox}

    \ExN{1} –í –ø–µ—Ä–≤–æ–π –∫–æ—Ä–æ–±–∫–µ 4 –±–µ–ª—ã—Ö –∏ 2 —á–µ—Ä–Ω—ã—Ö —à–∞—Ä–∞, –≤–æ –≤—Ç–æ—Ä–æ–π 1 –±–µ–ª—ã–π –∏ 2 —á–µ—Ä–Ω—ã—Ö. –ò–∑ –ø–µ—Ä–≤–æ–π –∫–æ—Ä–æ–±–∫–∏ –≤–æ –≤—Ç–æ—Ä—É—é –ø–µ—Ä–µ–ª–æ–∂–∏–ª–∏ 2 —à–∞—Ä–∞, –∑–∞—Ç–µ–º –∏–∑ –≤—Ç–æ—Ä–æ–π –∫–æ—Ä–æ–±–∫–∏ –¥–æ—Å—Ç–∞–ª–∏ —à–∞—Ä. –ö–∞–∫–æ–≤–∞
