    1. \textbf{–°–ø–µ—Ü–∏–∞–ª—å–Ω–∞—è –ø—Ä–∞–≤–∞—è —á–∞—Å—Ç—å}

    \Ex $y^{\prime\prime} - 3y^\prime + 3y = 2e^{3x} \quad (\heartsuit)$

    –ù–∞–≤–æ–¥—è—â–∏–µ —Å–æ–æ–±—Ä–∞–∂–µ–Ω–∏—è: –∑–∞–º–µ—Ç–∏–º, —á—Ç–æ $y = e^{ax}$ –Ω–µ –º–µ–Ω—è–µ—Ç —Å–≤–æ–π –≤–∏–¥ –ø—Ä–∏ –¥–∏—Ñ—Ñ–µ—Ä–µ–Ω—Ü–∏—Ä–æ–≤–∞–Ω–∏–∏,
    —Ç–∞–∫ –∂–µ –∫–∞–∫ –∏ $y = P_n(x)$, $y = A\cos bx + B\sin bx$

    –ò–º–µ–µ—Ç —Å–º—ã—Å–ª –∏—Å–∫–∞—Ç—å —á–∞—Å—Ç–Ω—ã–µ —Ä–µ—à–µ–Ω–∏—è –¥–ª—è $(\heartsuit)$ –≤ –≤–∏–¥–µ $y = Ae^{3x}$

    $(Ae^3x)^{\prime\prime} - 3(Ae^{3x})^\prime + 2Ae^{3x} = 2e^{3x}$

    $9A - 9A + 2A = 2 \Longrightarrow A = 1$, —Ç–æ –µ—Å—Ç—å $y^* = e^{3x}$

    \Nota –ï—Å–ª–∏ –ø—Ä–∞–≤–∞—è —á–∞—Å—Ç—å –õ–ù–î–£ —Å–æ–¥–µ—Ä–∂–∏—Ç –ø—Ä–æ–∏–∑–≤–µ–¥–µ–Ω–∏—è $e^{ax}, P_n(x), \cos bx, \sin bx$, —Ç–æ $y^*$ –∏—â–µ–º –≤ –≤–∏–¥–µ —Å–ø–µ—Ü–∏–∞–ª—å–Ω–æ–π –ø—Ä–∞–≤–æ–π —á–∞—Å—Ç–∏

    \Def –°–ø–µ—Ü–∏–∞–ª—å–Ω–∞—è –ø—Ä–∞–≤–∞—è —á–∞—Å—Ç—å (–°–ü–ß) -- —Ñ—É–Ω–∫—Ü–∏—è $f(x) = e^{ax} (P_n(x)\cos bx + Q_m (x)\sin bx)$ (–æ–±–æ–∑–Ω–∞—á–∏–º $k = a \pm ib$)

        \item $f(x) = P_n(x) e^{ax} \quad (b = 0)$

        \item $f(x) = A\cos b x + B \sin bx$ -- –≥–∞—Ä–º–æ–Ω–∏–∫–∞ $\quad (a = 0, n = m = 0)$

        \item $f(x) = P_n(x) \quad (a = b = 0)$
    \end{enumerate}

    –ú–µ—Ç–æ–¥: –†–µ—à–µ–Ω–∏–µ –∏—â–µ—Ç—Å—è –≤ –≤–∏–¥–µ $y^* = e^{ax} (\overline{P}_l \cos bx + \overline{Q}_l (x) \sin bx)$,
    –≥–¥–µ $a, b$ -- –∫–æ—ç—Ñ—Ñ–∏—Ü–∏–µ–Ω—Ç—ã –°–ü–ß, $l = \max(m, n), \overline{P}_l, \overline{Q}_l$ -- –º–Ω–æ–≥–æ—á–ª–µ–Ω—ã –≤ –Ω–µ–æ–ø—Ä–µ–¥–µ–ª–µ–Ω–Ω—ã—Ö –∫–æ—ç—Ñ—Ñ–∏—Ü–∏–µ–Ω—Ç–∞—Ö

    \ExN{1} $(\heartsuit)$: $y^{\prime\prime} - 3y^\prime + 3y = 2e^{3x} = e^{3x} (2 \cos 0x) \quad (k = 3 \pm 0 = 3)$

    $y^* = e^{3x} (\overline{P}_{l = 0} (x) \cos 0x) = e^{3x} \cdot A$

    \ExN{2} –û–¥–Ω–∞–∫–æ, –¥–ª—è —ç—Ç–æ–≥–æ —É—Ä–∞–≤–Ω–µ–Ω–∏—è: $y^{\prime\prime} - 3y^\prime + 2y = e^{2x}$

    –°–ü–ß: $e^{2x} = e^{2x} (1 \cos 0x + B \sin 0x) \quad k = a \pm ib = 2$

    \begin{rcases*}
    $y^* = Ae^{2x}$ \\
    $y^{*\prime} = 2Ae^{2x}$ \\
    $y^{*\prime\prime} = 4Ae^{2x}$ \\
    \end{rcases*} –î–£ $\longrightarrow \begin{matrix}4Ae^{2x} - 6Ae^{2x} + 2Ae^{2x} = e^{2x} \\ 4A - 6A + 2A = 1 \\ 0A = 1\end{matrix}$ -- –Ω–µ–ª—å–∑—è –Ω–∞–π—Ç–∏ $A$ {\Huge ü§Ø}

    –†–µ—à–∏–º –•—Ä–£: $\lambda_2 - 3\lambda + 2 = 0 \Longrightarrow \lambda_1 = 1, \lambda_2 = 2$

    –í —ç—Ç–æ–º —Å–ª—É—á–∞–µ —á–∏—Å–ª–æ $k$, —Å–æ–æ—Ç–≤–µ—Ç—Å—Ç–≤—É—é—â–µ–µ –°–ü–ß, —Å–æ–≤–ø–∞–¥–∞–µ—Ç —Å –∫–æ—Ä–Ω–µ–º –•—Ä–£

    \smallvspace

    –ò—Å—Å–ª–µ–¥—É–µ–º —Å–∏—Ç—É–∞—Ü–∏—é –Ω–∞ –ø—Ä–∏–º–µ—Ä–µ –°–ü–ß $f(x) = P_n(x) e^{ax}$

    –ü—É—Å—Ç—å –¥–∞–Ω–æ –î–£: $y^{\prime\prime} + py^\prime + qy = P_n(x)e^{ax}$

    –î–ª—è –Ω–µ–≥–æ –•—Ä–£: $\lambda^2 + p\lambda + q = 0 \Longrightarrow \lambda_{1,2}$ - –∫–æ—Ä–Ω–∏

    –ò—â–µ–º $y^* = \overline{P}_n(x) e^{ax}$

    $y^{*\prime} = \overline{P}_{n - 1} (x) e^{ax} + a\overline{P}_n(x) e^{ax}$

    $y^{*\prime\prime} = \overline{P}_{n - 2} (x) e^{ax} + 2a\overline{P}_{n - 1} (x) e^{ax} + a^2\overline{P}_n(x) e^{ax}$

    –ü–æ–ª—É—á–∞–µ–º:

    $\overline{P}_{n - 2} (x) e^{ax} + 2a\overline{P}_{n - 1} (x) e^{ax} + a^2\overline{P}_n(x) e^{ax} + (\overline{P}_{n - 1} (x) e^{ax} + a\overline{P}_n(x) e^{ax})p + \overline{P}_n(x) e^{ax} q = P_n(x) e^{ax}$

    $\overline{P}_{n - 2} (x) e^{ax} + (2a + p)\overline{P}_{n - 1} (x) e^{ax} + (a^2 + pa + q)\overline{P}_n(x) e^{ax} = P_n(x) e^{ax}$

    $\overline{P}_{n - 2} (x) + (2a + p)\overline{P}_{n - 1} (x) + (a^2 + pa + q)\overline{P}_n(x) = P_n(x)$

    \smallvspace

    –ó–∞–º–µ—Ç–∏–º, —á—Ç–æ –µ—Å–ª–∏ $a$ -- –∫–æ—Ä–µ–Ω—å –•—Ä–£, —Ç–æ –µ—Å—Ç—å $a \pm ib = a = k = \lambda_i$ (–ø—É—Å—Ç—å 1-–æ–π –∫—Ä–∞—Ç–Ω–æ—Å—Ç–∏), —Ç–æ $a^2 + pa + q = 0$ –∏ —Å—Ç–µ–ø–µ–Ω—å
    –ª–µ–≤–æ–π —á–∞—Å—Ç–∏ –ø–æ–Ω–∏–∂–∞–µ—Ç—Å—è –¥–æ $n - 1$

    –ï—Å–ª–∏ $a$ -- –∫–æ—Ä–µ–Ω—å –•—Ä–£ 2-–æ–π –∫—Ä–∞—Ç–Ω–æ—Å—Ç–∏, —Ç–æ –µ—Å—Ç—å $a^2 + pa + q = \left(a + \frac{p}{2}\right)^2 = 0 \Longleftrightarrow 2a + p = 0$, —Ç–æ —Å—Ç–µ–ø–µ–Ω—å –ª–µ–≤–æ–π —á–∞—Å—Ç–∏ –ø–æ–Ω–∏–∂–∞–µ—Ç—Å—è –Ω–∞ $2$

    –ß—Ç–æ–±—ã —Å–¥–µ–ª–∞—Ç—å —É—Ä–∞–≤–Ω–µ–Ω–∏–µ –¥–ª—è $\overline{P}_n$ —Ä–µ—à–∞–µ–º—ã–º, –¥–æ–º–Ω–æ–∂–∏–º $y^*$ –Ω–∞ $x^r$, –≥–¥–µ $r$ -- —á–∏—Å–ª–æ —Å–æ–≤–ø–∞–¥–µ–Ω–∏–π $k = a \pm ib$ —Å –∫–æ—Ä–Ω–µ–º –•—Ä–£ $\lambda_i$ (–¥—Ä—É–≥–∏–º–∏ —Å–ª–æ–≤–∞–º–∏, –∫—Ä–∞—Ç–Ω–æ—Å—Ç—å $\lambda_i$, —Å –∫–æ—Ç–æ—Ä—ã–º —Å–æ–≤–ø–∞–¥–∞–µ—Ç $k$)

    \mediumvspace

    –û–∫–æ–Ω—á–∞—Ç–µ–ª—å–Ω–æ, –º–µ—Ç–æ–¥ –≤—ã–≥–ª—è–¥–∏—Ç —Ç–∞–∫: –¥–ª—è —É—Ä–∞–≤–Ω–µ–Ω–∏—è $y^{\prime\prime} + py^\prime + qy = e^{ax} (P_n(x)\cos bx + Q_m (x)\sin bx)$, –≥–¥–µ $\lambda_{1,2}$ -- –∫–æ—Ä–Ω–∏ –•—Ä–£, $k = a \pm ib$

    –ß–∞—Å—Ç–Ω–æ–µ —Ä–µ—à–µ–Ω–∏–µ –≤—ã–≥–ª—è–¥–∏—Ç —Ç–∞–∫: \fbox{$y^* = x^r e^{ax} (\overline{P}_l (x)\cos bx + \overline{Q}_l (x) \sin bx$}, –≥–¥–µ $l = \max(m, n)$, $r$ - –∫—Ä–∞—Ç–Ω–æ—Å—Ç—å –∫–æ—Ä–Ω—è –•—Ä–£ $\lambda_i = k$

    \smallvspace

    –û–±–æ–±—â–∏–º –¥–ª—è –õ–î–£$_n$: $y^{(n)} + p_1 y^{(n - 1)} + \dots + p_n y = f(x)$

    –•—Ä–£ –≤—ã–≥–ª—è–¥–∏—Ç —Ç–∞–∫: $\lambda^n + p_1 \lambda^{n - 1} + \dots + p_n = 0$

    \mediumvspace

    \underline{–ü—Ä–∞–≤–∏–ª–æ} –ø–æ—Å—Ç—Ä–æ–µ–Ω–∏—è –§–°–† –¥–ª—è $\overline{y}$, –æ–±—â–µ–≥–æ —Ä–µ—à–µ–Ω–∏—è –æ–¥–Ω–æ—Ä–æ–¥–Ω–æ–≥–æ –î–£:

    \begin{enumerate}
        \item –í—Å—è–∫–æ–º—É $\lambda_i$ -- –æ–¥–∏–Ω–æ—á–Ω–æ–º—É $\Real$-–∫–æ—Ä–Ω—é –•—Ä–£ —Å–æ–ø–æ—Å—Ç–∞–≤–ª—è–µ–º $y_i = e^{\lambda_i x}$

        \item $\Real$-–∫–æ—Ä–Ω—é $\lambda$ –∫—Ä–∞—Ç–Ω–æ—Å—Ç–∏ $s$ —Å–æ–ø–æ—Å—Ç–∞–≤–ª—è–µ–º –Ω–∞–±–æ—Ä $\Set{y_1, y_2, \dots, y_s} = \Set{e^{\lambda x}, xe^{\lambda x}, \dots, x^{s - 1} e^{\lambda x}}$

        \item –í—Å—è–∫–æ–π –æ–¥–∏–Ω–æ—á–Ω–æ–π –ø–∞—Ä–µ $\lambda_{j_1,j_2} = \alpha_j \pm i\beta_j$ —Å–æ–æ—Ç–≤–µ—Ç—Å—Ç–≤—É–µ—Ç –ø–∞—Ä–∞ $\{e^{\alpha x} \cos\beta x, e^{\alpha x} \sin\beta x\}$

        \item –ö–æ–º–ø–ª–µ–∫—Å–Ω–æ–π –ø–∞—Ä–µ $\lambda = \alpha \pm i\beta$ –∫—Ä–∞—Ç–Ω–æ—Å—Ç–∏ $t$ —Å–æ–æ—Ç–≤–µ—Ç—Å—Ç–≤—É–µ—Ç –Ω–∞–±–æ—Ä 
        
        $\{e^{\alpha x} \cos \beta x, e^{\alpha x} \sin \beta x, x e^{\alpha x} \cos \beta x, \dots, x^{t - 1}e^{\alpha x} \cos\beta x, x^{t - 1}e^{\alpha x} \sin\beta x\}$
    \end{enumerate}

    \Nota –ö–æ–ª–∏—á–µ—Å—Ç–≤–æ –≤–µ–∫—Ç–æ—Ä–æ–≤ $y_i$ –≤ –§–°–† —Ä–∞–≤–Ω–æ –ø–æ—Ä—è–¥–∫—É $n$ –¥–∏—Ñ—Ñ–µ—Ä–µ–Ω—Ü–∏–∞–ª—å–Ω–æ–≥–æ —É—Ä–∞–≤–Ω–µ–Ω–∏—è

    –°–ø–µ—Ü–∏–∞–ª—å–Ω–∞—è –ø—Ä–∞–≤–∞—è —á–∞—Å—Ç—å –∏—â–µ—Ç—Å—è –≤ –≤–∏–¥–µ $y^* = x^r e^{ax} (P_n(x)\cos bx + Q_m (x)\sin bx)$, –≥–¥–µ $r$ - –∫—Ä–∞—Ç–Ω–æ—Å—Ç—å $\Real$-–∫–æ—Ä–Ω—è –∏–ª–∏ $\mathbb{C}$-–ø–∞—Ä—ã, —Å –∫–æ—Ç–æ—Ä—ã–º–∏ —Å–æ–≤–ø–∞–¥–∞–µ—Ç $k = a \pm ib$

    \Ex –í–µ—Ä–Ω–µ–º—Å—è –∫ $y^{\prime\prime} - 3y^\prime + 2y = e^{2x}$

    $\begin{rcases*}
    y^* = Ax^1 e^{2x} \\
    y^{*\prime} = Ae^{2x} 2Axe^{2x} \\
    y^{*\prime\prime} = 2Ae^{2x} + 2Ae^{2x} + 4Axe^{2x}
    \end{rcases*} \to (4 - 6 + 2) Axe^{2x} + (4 - 3) Ae^{2x} = e^{2x} \quad A = 1$

    $y(x) = C_1 e^{2x} + C_2 e^{2x} + xe^{2x}$

    \mediumvspace

    \item \textbf{–õ–∞–≥—Ä–∞–Ω–∂–∞}

    \Mems –†–µ—à–µ–Ω–∏–µ –õ–î–£$_1$: $y^\prime + py = f(x)$

    \begin{enumerate}[label*=\arabic*) ]
        \item –†–µ—à–∞–µ–º –õ–û–î–£ $y^\prime + py = 0$, –ø–æ–ª—É—á–∞–µ–º –§–°–† $\overline{y} = Cy_0$

        \item –†–µ—à–∞–µ–º –õ–ù–î–£, –∏—â–µ–º —Ä–µ—à–µ–Ω–∏—è –≤ –≤–∏–¥–µ $y(x) = C(x)y_0$, –ø–æ–ª—É—á–∞–µ–º $C^\prime(x) y_0 = f(x) \to C(x)$
    \end{enumerate}

    \Nota –í–≤–µ–¥–µ–º –∞–Ω–∞–ª–æ–≥–∏—á–Ω—ã–π –º–µ—Ç–æ–¥ –¥–ª—è –õ–î–£$_2$:

    \begin{enumerate}[label*=\arabic* —ç—Ç–∞–ø) ]
        \item $y^{\prime\prime} + py^\prime + qy = 0$ -- –õ–û–î–£, $\lambda_{1, 2}$ -- –∫–æ—Ä–Ω–∏, —Å–æ–æ—Ç–≤–µ—Ç—Å—Ç–≤—É—é—â–∏–µ –§–°–† $\{y_1, y_2\}$

        –ü–æ–ª—É—á–∞–µ–º –æ–±—â–µ–µ —Ä–µ—à–µ–Ω–∏–µ $\overline{y}(x) = C_1 y_1 + C_2 y_2$

        \item –í–∞—Ä—å–∏—Ä—É–µ–º $C_1$ –∏ $C_2$, –Ω–æ —Ç–µ–ø–µ—Ä—å –Ω—É–∂–Ω—ã –¥–≤–∞ —É—Å–ª–æ–≤–∏—è –¥–ª—è –∏—Ö –æ–ø—Ä–µ–¥–µ–ª–µ–Ω–∏—è. –û–¥–Ω–∏–º –∏–∑ –Ω–∏—Ö —è–≤–ª—è–µ—Ç—Å—è —Å–∞–º–æ –î–£
    \end{enumerate}

    \Ex $y^{\prime\prime} - 3y^\prime + 2y = 2e^{3x}$

    $\overline{y} = C_1 e^x + C_2 e^{2x}$

    $y(x) = C_1 e^x + C_2 e^{2x} + y^*$

    $(g(x) + C_1)e^x + (h(x) + C_2)e^{2x} = C_1 e^x + C_2 e^{2x} + g(x)e^x + h(x)e^{2x}$

    –ü–æ–¥–±–µ—Ä–µ–º $g, h$: $\underset{g}{\undergroup{\frac{e^{2x}}{2}}}e^x + \underset{h}{\undergroup{\frac{e^x}{2}}}e^{2x} = e^{3x}$ –∏–ª–∏
    $\underset{g}{\undergroup{-e^{2x}}} e^x + \underset{h}{\undergroup{2e^x}} e^{2x} = e^{3x}$

    –ó–∞–º–µ—Ç–∏–º, —á—Ç–æ $C_1^\prime(x)$ –≤–æ –≤—Ç–æ—Ä–æ–º —Å–ª—É—á–∞–µ $g^\prime = -2e^{2x}$, –∞ $C_2^\prime = 2e^x$

    –¢–æ–≥–¥–∞ $C_1^\prime(x) e^x + C^\prime_2 (x) e^{2x} = -2e^{3x} + 2e^{3x} = 0$

    \Nota –ü–æ–¥—Å—Ç–∞–≤–∏–º $y(x) = C_1 (x) y_1 + C_2 (x) y_2$ –≤ –î–£

    –ü–æ–ª—É—á–∞–µ–º –ø—Ä–æ–∏–∑–≤–æ–¥–Ω—É—é $y^\prime(x) = C^\prime_1(x) y_1 + C_1(x)y^\prime_1 + C^\prime_2(x) y_2 + C_2(x)y^\prime_2$

    –¢—Ä–µ–±—É–µ–º $C^\prime_1 y_1 + C_2^\prime y_2 = 0$

    –¢–æ–≥–¥–∞ –≤—Ç–æ—Ä–∞—è –ø—Ä–æ–∏–∑–≤–æ–¥–Ω–∞—è: $y^{\prime\prime}(x) = C_1^\prime (x) y_1^\prime + C_1 (x) y_1^{\prime\prime} + C_2^\prime (x) y_2^\prime + C_2 (x) y_2^{\prime\prime}$

    $C_1^\prime (x) y_1^\prime + C_1 (x) y_1^{\prime\prime} + C_2^\prime (x) y_2^\prime + C_2 (x) y_2^{\prime\prime} + pC_1(x)y^\prime_1 + pC_2(x)y^\prime_2 + qC_1(x)y_1 + qC_2(x)y_2 = f(x)$

    $\underset{=\ 0}{C_1(x)Ly_1} + \underset{=\ 0}{C_2(x)Ly_2} + C_1^\prime(x)y_1^\prime + C_2^\prime(x)y_2^\prime = f(x)$

    \underline{–ò—Ç–∞–∫}, —Å–∏—Å—Ç–µ–º–∞ –¥–ª—è –æ–ø—Ä–µ–¥–µ–ª–µ–Ω–∏—è $C_1(x), C_2(x)$:
    \begin{cases}
        C^\prime_1(x)y_1 + C^\prime_2(x)y_2 = 0 \\
        C_1^\prime(x)y_1^\prime + C_2^\prime(x)y_2^\prime = f(x)
    \end{cases}
